namespace nskr{
    int i, j;
    struct unionfind{
        //unionfind - union by size, 経路圧縮
        //Library Checker - Unionfind - 29ms 2.30Mib
        //https://judge.yosupo.jp/submission/369440
        vector pr, sz;
        unionfind(int n){
            pr.resize(n);
            sz.resize(n,1);
            rep(i,n) pr[i] = i;
        }

        inline int find(int x){ 
            while(pr[x] != x){ 
                pr[x] = pr[pr[x]];
                x = pr[x];
            }
            return x;
        }
        
        inline void merge(int p, int q){ 
            p = find(p); q = find(q);
            if(p==q) return ;
            if(sz[p] < sz[q]) swap(p,q);
            pr[q] = p;
            sz[p] += sz[q];
            return ;
        }

        inline bool same(int p, int q){
            if(find(p) == find(q)) return true;
            return false;
        }

        int size(int x){
            return sz[find(x)];
        }
    };
}

unionfind uf(int n);

n頂点0辺の無向グラフを作る。
この時のすべての頂点において、代表元は自分自身

制約　 : vectorと同じ
計算量 : \( O(n) \)

  • unionfind with potential submission 

namespace nskr{
    int i, j;

    template <class S, S (*op)(S,S), S (*inv)(S), S (*e)()>
    struct unionfind_p{
        //unionfind (with potential) - union by size, 経路圧縮
        //Library Checker - Unionfind with Potential (Non-Commutative Group) - 86ms / 7.99Mib
        //https://judge.yosupo.jp/submission/369371
        // S には operator!= が定義されている必要がある
        // op : 積 / inv : 逆元 / e : 単位元
        vector<int> pr, sz;
        vector<S> val;
        unionfind_p(int n){
            pr.resize(n);
            sz.resize(n,1);
            val.resize(n,e()); 
            rep(i,n) pr[i] = i;
        }

        pair<int,S> find(int x){ 
            if(x == pr[x]) return {x,e()};
            pair<int,S> y = find(pr[x]);
            val[x] = op(val[x],y.second);
            pr[x] = y.first;
            return {pr[x],val[x]};
        }
        
        bool merge(int p, int q, S x){ 
            pair<int, S> P = find(p), Q = find(q);
            if(P.first==Q.first){
                if(P.second != op(x,Q.second)) return false; //invalid;
                return true; //valid
            }
            if(sz[P.first] < sz[Q.first]) {swap(P,Q); x = inv(x);}
            pr[Q.first] = P.first;
            val[Q.first] = op(op(inv(Q.second), inv(x)),P.second);
            sz[p] += sz[q];
            return true; //valid
        }

        pair<bool,S> get(int p, int q){
            pair<int, S> P = find(p), Q = find(q);
            if(P.first == Q.first){
                return {true,op(P.second,inv(Q.second))};
            }
            return {false,e()};
        }
    };
}

＊概要

群、つまり

・集合 \(S\)
・S上の二項演算 \( \cdot \)
・逆元 \(^{-1}\)
・単位元 \(e\)
があって、
・結合法則　\( \forall a,b,c \in S,\)　\( (a\cdot b)\cdot c = a\cdot(b\cdot c)\)
・単位元の存在　\( \forall a \in S,　 a \cdot e = e\cdot a = a\)
・逆元の存在　\( \forall a \in S,　 \exists a^{-1}\in S,　 a \cdot a^{-1} = a^{-1} \cdot a = e \)
の3つを満たす代数構造に対し使用できる。

※Sはoperator!=が実装されている必要がある

↑絶対忘れるのででかく書いた

ポテンシャル付き有効辺をもらってmergeを行う　有向っぽいグラフを作る（逆向きに進むとその辺のポテンシャルの逆元がかけられる）
この構造において、適当な頂点xからyへの経路がいくつかあったとして、その経路の取り方に依らず、通った辺の値の総和（総積）は等しい
実装上は各頂点がポテンシャルを持ってて、その値は自分の親とつながる辺のポテンシャルに等しい

あと、計算量はop(),inv(),e()が\(O(1)\)である前提 でなければたぶん大体その分だけ倍

＊コンストラクタ

unionfind_p<S, op, inv, e> uf(int n);

\(S\) 乗せる要素の型
\(op(S \space a,S \space b)\) 二項演算
\(inv(S \space a)\) aの逆元
\(e()\) 単位元
要素数nで初期化、この時すべての頂点の代表元は自分自身であり、持つポテンシャルは単位元

制約　 : vectorと同じ
計算量 : \( O(n) \)

＊find

pair<int,S> uf.find(int x)

頂点xが属する連結成分の代表元と、xからそこまでの経路のポテンシャルの総和を返す

制約　 : \( 0\leq x < n \)
計算量 : ならし\( O(α(n)) \)

＊merge

bool uf.merge(int p, int q, S x)

頂点p,q間にp→qの向きでポテンシャルxをもつ辺を張る。（実際に辺が張られるのはそれぞれの代表元の間） また、

・新しい辺のポテンシャルが、これまでの辺のポテンシャルと矛盾するならば false
・でなければ true
を返す。もともと同じ代表元を持つ2点間は矛盾するかの判定だけやる

    制約　 : \( 0\leq p,q < n \)
    計算量 : ならし\( O(α(n)) \)

    ＊get

    pair<bool, S> uf.get(int p, int q)

    経路p→qの通る辺のポテンシャルの総積を返す。経路が存在しない場合boolがfalseで帰ってくる　勿論あればtrue

    制約　 : \( 0\leq p,q < n \)
    計算量 : ならし\( O(α(n)) \)
